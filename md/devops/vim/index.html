    # =========================
# DOSYA AÇMA / KAPAMA / KAYDETME
# =========================

vi a.txt b.txt c.txt              # a.txt, b.txt, c.txt aynı vi oturumunda aç (argüman listesi)
:args                             # mevcut oturumda yüklü dosyaları listele
:n                                # argüman listesindeki SONRAKİ dosyaya geç (a.txt -> b.txt)
:N                                # argüman listesindeki ÖNCEKİ dosyaya geç
:wn                               # mevcut dosyayı kaydet + argüman listesindeki bir sonrakine geç
:wq                               # kaydet + vi'den/Vim'den çık
:q!                               # kaydetmeden zorla çık (geri alınamaz, dikkat)

:e a.txt                          # içerideyken a.txt dosyasını aç (aktif dosyayı değiştir)
:e! a.txt                         # kaydedilmemiş değişiklikleri atarak a.txt'yi zorla yükle
:r b.txt                          # b.txt içeriğini mevcut satırın ALTINA ekle
:r !ls                            # 'ls' çıktısını mevcut satırın ALTINA ekle (ex komutu, vi/Vim)
:w                                # mevcut dosyayı diske yaz
:w yeni_ad.txt                    # mevcut içeriği yeni_ad.txt adıyla kaydet (save as)
:wn | q                           # kaydet + sonraki dosyaya geç + sonra çık (bazı vi/vim sürümlerinde)

# =========================
# SATIR / DOSYA İÇİ GEZİNME
# =========================

h / j / k / l                     # sola / aşağı / yukarı / sağa hareket
0 / ^ / $                         # satır başı / ilk boşluk olmayan karakter / satır sonu
- / +                             # bir önceki / bir sonraki satırın ilk non-blank karakteri

w / 2w / 10w                      # 1 / 2 / 10 kelime ileri git
b / 2b / 10b                      # 1 / 2 / 10 kelime geri git
e / 2e                            # 1 / 2 kelime sonuna kadar ilerle

( / )                             # bir önceki / bir sonraki cümlenin başına git
{ / }                             # bir önceki / bir sonraki paragrafın başına git

G                                 # dosya sonuna git
1G                                # dosya başına git (1. satır)
10G                               # 10. satıra git
42G                               # 42. satıra git (örnek senaryo: hata satırına atlamak)

H / M / L                         # ekranın üst / orta / alt satırına git (High/Middle/Low)
Ctrl+u / Ctrl+d                   # yarım ekran yukarı / yarım ekran aşağı
Ctrl+b / Ctrl+f                   # bir ekran geri / bir ekran ileri (back/forward)

f; / f( / f{                      # satır içinde ';' / '(' / '{' karakterine kadar git
t;                                # ';' karakterinin bir ÖNCEKİ karakterine git
; / ,                             # son f/t komutunu aynı yönde / ters yönde tekrar et
%                                 # parantez üzerindeyken eşleşen paranteze zıpla ((), {}, [] içinde çok kullanışlı)

# =========================
# INSERT / DEĞİŞTİRME MODLARI
# =========================

i                                 # imlecin soluna metin ekle (insert mode)
I                                 # satır başına (ilk non-blank) git + eklemeye başla
a                                 # imlecin SAĞINA ekle
A                                 # satır SONUNA git + eklemeye başla

o                                 # alt satıra yeni satır aç + eklemeye başla
O                                 # üst satıra yeni satır aç + eklemeye başla

rX                                # imlecin altındaki karakteri X ile değiştir (tek karakter overwrite)
R                                 # replace mode: yazdıkça var olan karakterlerin üzerine yaz
Esc                               # insert/replace moddan çık, normal moda dön

# (Vim) gi                        # son insert pozisyonuna dön + insert moda gir (tekrar düzenleme için faydalı)

# =========================
# SİLME / KOPYALAMA / YAPIŞTIRMA (OPERATOR + MOTION)
# =========================

x / X                             # imlecin altındaki karakteri / solundaki karakteri sil
dw / d2w                          # bir kelimeyi / iki kelimeyi sil (delete word)
db / de                           # kelime başına / kelime sonuna kadar sil
d$ / d0                           # imleçten satır sonuna / satır başına kadar sil
dG / d1G                          # bulunduğun satırdan dosya sonuna / dosya başına kadar sil
d) / d}                           # cümle sonuna / paragraf sonuna kadar sil

dd / 2dd / 5dd                    # 1 / 2 / 5 satırı sil
D                                 # imleçten satır sonuna kadar sil (d$ ile eşdeğer)

cw / c2w                          # bir kelimeyi / iki kelimeyi değiştir (sil + insert)
c$ / c0                           # satır sonuna / satır başına kadar değiştir
cc                                # satırın tamamını değiştir (satırı temizle + insert mode)
C                                 # imleçten satır sonuna kadar değiştir (c$ ile eşdeğer)

yy / 2yy / 5yy                    # 1 / 2 / 5 satırı yank (kopyala)
y$ / y0                           # imleçten satır sonuna / başına kadar yank
yG / y1G                          # bulunduğun satırdan dosya sonuna / başına kadar yank

p / P                             # imleçten SONRA / ÖNCE yapıştır
]p / [p                           # (Vim) indent ayarlı yapıştırma (özellikle kodda kullanışlı)

> / <                             # seçili scope'u sağa / sola indent et (motion veya visual ile birlikte)
>> / <<                           # tüm satırı sağa / sola indent et
>G / <G                           # bulunduğun satırdan dosya sonuna kadar indent / unindent

.                                 # son düzenleme komutunu tekrar et (çok kritik verimlilik komutu)
u                                 # son değişikliği geri al (undo)
U                                 # satırı önceki haline döndür (satır bazlı undo davranışı)

# =========================
# ARAMA / BULMA
# =========================

/pattern                          # ileri yönde pattern ara
?pattern                          # geri yönde pattern ara
n / N                             # aynı pattern için bir SONRAKİ / BİR ÖNCEKİ eşleşmeye git

/\<foo\>                          # tam kelime olarak 'foo' ara (başında/sonunda harf veya rakam yok)
:/^main/                          # 'main' ile başlayan satıra git
:/error/                          # 'error' içeren sonraki satıra git

# (Vim) * / #                     # imlecin altındaki kelimeyi ileri / geri yönde ara
# (Vim) :nohlsearch               # arama highlight'ını kapat

# =========================
# ARAMA + DEĞİŞTİRME (:s)
# =========================

:s/old/new/                       # bulunduğun satırda ilk 'old' -> 'new'
:s/old/new/g                      # bulunduğun satırda TÜM 'old' -> 'new'
:%s/old/new/g                     # tüm dosyada 'old' -> 'new'
:%s/old/new/gc                    # tüm dosyada değiştir, her seferinde onay iste (confirm)
:10,20s/foo/bar/g                 # 10–20 satır arası foo -> bar
:.,$s/foo/bar/g                   # bulunduğun satırdan dosya SONUNA kadar foo -> bar
:%s/\<int\>/int32_t/g             # sadece kelime olarak 'int' geçen yerleri 'int32_t' yap

# sonu boşluk olan satırları temizleme (trailing whitespace):
:%s/[ \t]\+$//                    # satır sonundaki boşluk ve tab'leri sil

# =========================
# :global / :v İLE TOPLU İŞLEM
# =========================

:g/TODO/p                         # TODO içeren tüm satırları ekrana yaz
:g/TODO/d                         # TODO içeren tüm satırları sil
:g/^#/d                           # '#' ile başlayan (yorum) satırların hepsini sil
:g/^$/d                           # boş satırları sil

:v/DEBUG/d                        # DEBUG içermeyen satırları sil (sadece DEBUG satırları kalsın)
:g/^/s/$/;/                       # her satırın SONUNA ';' ekle (tüm satırlara substitution uygula)

# (Vim) :g/pattern/normal A;      # pattern içeren tüm satırlarda normal modda 'A;' çalıştır (satır sonuna ';' ekler)

# =========================
# MARKLAR (İŞARETLER) VE ATLAMALAR
# =========================

ma                                # bulunduğun satıra 'a' markını koy
mb                                # bulunduğun satıra 'b' markını koy
'a                                # 'a markının satırına git
`a                                # 'a markının satır+kolon pozisyonuna git

''                                # bir önceki jump konumuna satır bazında geri dön
``                                # (Vim) bir önceki konuma tam pozisyonla dön

# Örnek senaryo:
# ma                                # fonksiyon başlangıcına 'a' mark koy
# G                                 # dosya sonuna git
# 'a                                # tekrar fonksiyon başına dön

# =========================
# REGISTER (BUFFER) KULLANIMI
# =========================

"ayy                              # satırı 'a' register'ına yank et
"byw                              # kelimeyi 'b' register'ına yank et
"ap                               # 'a' register'ını yapıştır
"bdd                              # satırı sil ve 'b' register'ına koy (cut)
"bp                               # 'b' register'ını yapıştır

"0p                               # son "normal yank" register'ını yapıştır (silmeden yank'lenen)
"1p                               # son silinen (cut) içeriği yapıştır (implementasyona göre)

# (Vim) "+y                        # sistem clipboard'a kopyala (clipboard destekliyse)
# (Vim) "+p                        # sistem clipboard'dan yapıştır

# =========================
# MAKRO KULLANIMI (Vim)
# =========================

qa                                # 'a' register'ına makro kaydını başlat
# ... burada yapmak istediğin komutları sırayla çalıştır ...
q                                 # kaydı bitir
@a                                # 'a' makrosunu çalıştır
@@                                # son kullanılan makroyu tekrar çalıştır
10@a                              # 'a' makrosunu 10 kez çalıştır (ör: 10 satırı aynı şekilde düzenle)

# =========================
# PENCERE / SPLIT YÖNETİMİ (Vim)
# =========================

:split                            # yatay split: ekranı yatay böl, aynı dosyayı 2 pencere göster
:vsplit                           # dikey split: ekranı dikey böl
:split b.txt                      # altta b.txt dosyasını aç
:vsplit c.txt                     # sağda c.txt dosyasını aç

Ctrl+w h / j / k / l              # pencere (split) odaklarını sola / aşağı / yukarı / sağa taşı
Ctrl+w w                          # açık pencereler arasında sırayla dolaş
Ctrl+w =                          # tüm pencerelerin boyutlarını eşitle
Ctrl+w + / -                      # aktif pencerenin yüksekliğini artır / azalt
Ctrl+w > / <                      # aktif pencerenin genişliğini artır / azalt

:q                                # AKTİF split'i kapat (diğer pencereler açık kalır)
:close                            # aktif pencereyi kapat (q ile benzer)
Ctrl+w c                          # aktif pencereyi kısayol ile kapat
:only                             # sadece aktif pencere kalsın, diğer tüm pencereleri kapat
Ctrl+w o                          # :only ile aynı davranış

# =========================
# BUFFER YÖNETİMİ (Vim)
# =========================

:ls                               # açık buffer listesini göster
:badd a.txt                       # a.txt dosyasını buffer listesine ekle (ekranda açmadan)
:badd src/*.c                     # glob pattern ile tüm .c dosyalarını buffer listesine ekle
:buffer 1                         # 1 numaralı buffer'a geç (örneğin a.txt)
:bnext / :bprev                   # sonraki / önceki buffer'a geç
:bd                               # aktif buffer'ı kapat (delete buffer)
:bd 3                             # 3 numaralı buffer'ı kapat

# Örnek senaryo:
# vim a.txt
# :badd b.txt                     # b.txt arkada hazır
# :badd c.txt                     # c.txt arkada hazır
# :vsplit                         # dikey split aç
# :buffer 2                       # bu split'te b.txt'yi göster
# :split                          # bu split'i yatay böl
# :buffer 3                       # alt split'te c.txt'yi göster

# =========================
# TAB YÖNETİMİ (Vim)
# =========================

:tabnew                           # yeni sekme aç (boş)
:tabnew b.txt                     # b.txt'yi yeni sekmede aç
:tabnext / :tabprev               # sonraki / önceki sekmeye geç
:tabclose                         # aktif sekmeyi kapat
:tabs                             # tüm sekmeleri listele

# =========================
# DIŞ KOMUTLAR İLE ENTEGRASYON
# =========================

:!make                            # shell'de 'make' çalıştır, sonucu göster
:!ls                              # klasör içeriğini göster (shell komutu)
:!git status                      # git durumunu göster (shell üzerinden)

:r !dmesg | tail -20              # dmesg'in son 20 satırını mevcut dosyaya ekle
:w !sudo tee %                    # dosyayı sudo ile yaz (root izinine ihtiyaç varsa, Vim tarafında çok kullanılır)

# =========================
# TEMEL AYARLAR (Vim AĞIRLIKLI)
# =========================

:set nu                           # satır numaralarını aç (number)
:set nonu                         # satır numaralarını kapat
:set list                         # tab/boşluk gibi görünmeyen karakterleri göster
:set nolist                       # görünmeyen karakter görünümünü kapat
:set ts=4 sw=4                    # tabstop=4, shiftwidth=4 (indent davranışı)
:set noexpandtab                  # tab tuşu gerçek TAB karakteri yazsın
:set expandtab                    # tab tuşu boşluk yazsın

:set ignorecase                   # aramalarda case-insensitive
:set noignorecase                 # aramalarda case-sensitive
:set hlsearch                     # aranan pattern'leri highlight et
:set nohlsearch                   # arama highlight'ını kapat
